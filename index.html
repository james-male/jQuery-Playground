Toggle Content

Logic

  All show and hide functions are selected based on class names
  • Include jQuery Library (In this example I am using a google CDN link)
  • Default content is shown and button content hidden using an inline CSS when the page is loaded
  • On button click
    • Hide content for the buttons that were not clicked as well as the default content
    • Show content for the button that was clicked
  • On Reset click
    • Hide content from all buttons
    • Show default content

Code


HTML

// Include jQuery
<script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script>

// Buttons
<button type="button" class="button-default">Default</button>
<button type="button" class="button-left">Left</button>
<button type="button" class="button-left">Middle</button>
<button type="button" class="button-left">Right</button>

// Content
<p class="default-content">Default Content</p>
<p class="left-content">Left Button Content</p>
<p class="middle-content">Middle Button Content</p>
<p class="right-content">Right Button Content</p>

jQuery

$(document).ready(function(){
// Button Left
 $(".button-left").click(function(){
$(".default-content").hide();
$(".middle-content").hide();
$(".right-content").hide();
$(".left-content").show();
});

// Button Middle
 $(".button-middle").click(function(){
$(".default-content").hide();
$(".left-content").hide();
$(".right-content").hide();
$(".middle-content").show();
});

// Button Right
 $(".button-right").click(function(){
$(".default-content").hide();
$(".left-content").hide();
$(".middle-content").hide();
$(".right-content").show();
});

// Default
 $(".button-default").click(function(){
$(".left-content").hide();
$(".middle-content").hide();
$(".right-content").hide();
$(".default-content").show();
});
});
